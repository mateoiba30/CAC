ORG 1000H
CAR DB "p"
VOCALES DB "aeiouAEIOU"
AUX DB ?

ORG 3000H; DL=0FFH si el caracter es una vocal, o 00H en caso contrario.
ES_VOCAL:CMP [BX], AL
JNZ SALTO
MOV DL, 0FFH
JMP FIN
SALTO:INC BX
DEC CL
JNZ ES_VOCAL
FIN:RET

ORG 2000H;CAR POR VALOR
MOV AL, CAR
MOV DL, 0
MOV BX, OFFSET VOCALES
MOV CL, OFFSET AUX - OFFSET VOCALES
CALL ES_VOCAL
HLT
END
